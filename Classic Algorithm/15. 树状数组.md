 树状数组：查询和更新的时间复杂度都是O(logn)  

# 基本概念
  - 查询和更新都是 O(log n)
  - 快速计算一段连续子数组的元素和
  - 快速修改数组元素且维护性质
  - 初始化树状数组：下标范围 [1, n]（或 0-based 偏移）
  - 拆分区间形成关键区间（由 lowbit 决定）

# Lowbit 定义
  lowbit(i) = i & -i：保留二进制最低位的 1 及其后面的 0  

# 关键区间表示
  - 每个下标 i 管理的区间为 [i - lowbit(i), i]
  - 所有下标管理的区间右端点互不相同，覆盖整个数组

# 初始化
  1. 使用 update() 方法初始化：  
  ```java
  private int[] tree;   // 树状数组，下标 1..n
  private int[] nums;   // 原始数组副本（用于计算 delta）
  private int n;        // 数组长度

  // 构造方法1：O(n log n) 初始化
  public FenwickTree(int[] nums) {
      this.n = nums.length;
      this.nums = new int[n];
      this.tree = new int[n + 1];
      for (int i = 0; i < n; i++) {    // 每个nums[i]调用1次update(int index, int val)
          update(i, nums[i]);
      }
  }
  ```

  2. 使用 update() 方法初始化：
  ```java
 // 构造方法2：O(n) 初始化（更高效）

      this.n = nums.length;
  ```

# 完整代码实现
  ```java
  import java.util.List;
  
  /**
   * 树状数组（Fenwick Tree）实现
   * 支持单点更新、前缀和查询、区间和查询
   * 内部使用 1-based 索引，对外提供 0-based 接口
   */
  public class FenwickTree {
      private int[] tree;   // 树状数组，下标 1..n
      private int[] nums;   // 原始数组副本（用于计算 delta）
      private int n;        // 数组长度
  
      // 构造方法1：O(n log n) 初始化
      public FenwickTree(int[] nums) {
          this.n = nums.length;
          this.nums = new int[n];
          this.tree = new int[n + 1];
          for (int i = 0; i < n; i++) {
              update(i, nums[i]);
          }
      }
  
      // 构造方法2：O(n) 初始化（更高效）
      public FenwickTree(int[] nums, boolean linearInit) {
          this.n = nums.length;
          this.nums = nums.clone();
          this.tree = new int[n + 1];
          // 等价于 Python 的 O(n) 构建方式
          for (int i = 1; i <= n; i++) {
              tree[i] += nums[i - 1];
              int nxt = i + lowbit(i);
              if (nxt <= n) {
                  tree[nxt] += tree[i];
              }
          }
      }
  
      // 单点更新：将下标 index 处的值设置为 val
      public void update(int index, int val) {
          int delta = val - nums[index];
          nums[index] = val;
          int i = index + 1;          // 转为 1-based
          while (i <= n) {
              tree[i] += delta;
              i += lowbit(i);
          }
      }
  
      // 查询前缀和 [0, index] （闭区间）
      public int prefixSum(int index) {
          int i = index + 1;          // 转为 1-based
          int sum = 0;
          while (i > 0) {
              sum += tree[i];
              i -= lowbit(i);
          }
          return sum;
      }
  
      // 查询区间和 [left, right] （闭区间）
      public int rangeSum(int left, int right) {
          if (left > right) return 0;
          return prefixSum(right) - (left == 0 ? 0 : prefixSum(left - 1));
      }
  
      // lowbit 函数
      private int lowbit(int x) {
          return x & -x;
      }
  
      // 测试示例
      public static void main(String[] args) {
          int[] arr = {1, 3, 5, 7, 9};
          FenwickTree ft = new FenwickTree(arr, true);  // O(n) 构建
  
          System.out.println("前缀和 [0..2]：" + ft.prefixSum(2)); // 1+3+5=9
          System.out.println("区间和 [1..3]：" + ft.rangeSum(1, 3)); // 3+5+7=15
  
          ft.update(2, 6);  // 将 arr[2] 从 5 改为 6
          System.out.println("修改后区间和 [1..3]：" + ft.rangeSum(1, 3)); // 3+6+7=16
      }
  }
  ```
