# 定义
  s[0] = 0  
  s[i] = a[0] + ... + a[i-1]  
  s[i+1] = a[0] + ... + a[i-1] + a[i]  

# 性质
  a[i] + a[i+1] + ... + a[j] = s[j+1] - s[i]  

# 初始化前缀和数组
  s[i] =  
  - 0,                i = 0
  - s[i-1] + a[i-1],    i > 0
  
  ```java
  // 假设原数组 a 已定义，长度为 n
  int n = a.length;
  int[] s = new int[n + 1]; // s[0] 自动初始化为 0
  
  // 迭代计算前缀和
  for (int i = 1; i <= n; i++) {
      s[i] = s[i - 1] + a[i - 1]; // a 的索引从 0 开始
  }
  ```

# 求区间和
  c = s[j+1] - s[i]    //  a[i, j] 区间和
