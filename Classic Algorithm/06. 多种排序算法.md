为什么最常使用①快速排序呢？
- 常数因子①< ②
- 堆顶出堆后（堆顶与最末元素交换），需要去建堆

# 快排
  ```java
  public class QuickSort {
  
      /**
       * 快速排序入口
       * @param arr   待排序数组
       * @param left  左边界（包含）
       * @param right 右边界（包含）
       */
      public static void quickSort(int[] arr, int left, int right) {
          if (left >= right) {
              return;
          }
          int pivotIndex = partition(arr, left, right);    // 快排，返回新基轴
          quickSort(arr, left, pivotIndex - 1);            // 递归快排左半
          quickSort(arr, pivotIndex + 1, right);           // 递归快排右半
      }
  
      /**
       * 分区函数（Lomuto 分区方案）
       * @param arr   数组
       * @param left  左边界
       * @param right 右边界
       * @return 基准元素的最终位置
       */
      private static int partition(int[] arr, int left, int right) {
          // 选择中间元素作为基准，并将其交换到最右端
          int mid = left + (right - left) / 2;
          swap(arr, mid, right);
          int pivot = arr[right];
  
          int i = left;   // i 指向下一个小于基准的元素应该放置的位置
          for (int j = left; j < right; j++) {
              if (arr[j] < pivot) {
                  swap(arr, i, j);
                  i++;
              }
          }
          // 将基准放到正确位置（i 位置）
          swap(arr, i, right);
          return i;
      }
  
      private static void swap(int[] arr, int i, int j) {
          int tmp = arr[i];
          arr[i] = arr[j];
          arr[j] = tmp;
      }
  }
  ```

# 归并
  ```java
  public class MergeSort {
  
      /**
       * 归并排序入口
       * @param arr   待排序数组
       * @param left  左边界（包含）
       * @param right 右边界（包含）
       */
      public static void mergeSort(int[] arr, int left, int right) {
          if (left < right) {
              int mid = left + (right - left) / 2;
              mergeSort(arr, left, mid);            // 递归归并左半
              mergeSort(arr, mid + 1, right);       // 递归归并右半
              merge(arr, left, mid, right);         // 归并
          }
      }
  
      /**
       * 合并两个已排序的子数组 [left, mid] 和 [mid+1, right]
       */
      private static void merge(int[] arr, int left, int mid, int right) {
          // 临时数组存放合并结果
          int[] temp = new int[right - left + 1];
          int i = left;      // 左子数组起始索引
          int j = mid + 1;   // 右子数组起始索引
          int k = 0;         // 临时数组索引
  
          // 比较两个子数组的元素，将较小的放入临时数组
          while (i <= mid && j <= right) {
              if (arr[i] <= arr[j]) {
                  temp[k++] = arr[i++];
              } else {
                  temp[k++] = arr[j++];
              }
          }
  
          // 处理左子数组剩余元素
          while (i <= mid) {
              temp[k++] = arr[i++];
          }
  
          // 处理右子数组剩余元素
          while (j <= right) {
              temp[k++] = arr[j++];
          }
  
          // 将临时数组中的元素复制回原数组
          for (int p = 0; p < temp.length; p++) {
              arr[left + p] = temp[p];
          }
      }
  }
  ```

# 堆排序
  ```java
  public class HeapSort {
  
      /**
       * 堆排序入口
       * @param arr 待排序数组
       */
      public static void heapSort(int[] arr) {
          int len = arr.length;
  
          // 1. 建堆：从最后一个非叶子节点开始向下调整
          for (int i = len / 2 - 1; i >= 0; i--) {
              heapify(arr, len, i);
          }
  
          // 2. 交换堆顶与末尾，并缩小堆范围，重新调整
          for (int i = len - 1; i > 0; i--) {
              swap(arr, 0, i);       // 堆顶（最大）移到末尾
              heapify(arr, i, 0);    // 对剩余堆进行调整
          }
      }
  
      /**
       * 堆化（调整）函数：将以 i 为根的子树调整为最大堆
       * @param arr 数组
       * @param n   堆的大小（有效元素个数）
       * @param i   当前根节点下标
       */
      private static void heapify(int[] arr, int n, int i) {
          int largest = i;          // 假设当前节点最大
          int left = 2 * i + 1;     // 左孩子下标
          int right = 2 * i + 2;    // 右孩子下标
  
          // 如果左孩子存在且大于当前最大节点，更新 largest
          if (left < n && arr[left] > arr[largest]) {
              largest = left;
          }
          // 如果右孩子存在且大于当前最大节点，更新 largest
          if (right < n && arr[right] > arr[largest]) {
              largest = right;
          }
  
          // 如果最大节点不是 i，则交换根节点和largest节点，并递归调整受影响的子树
          if (largest != i) {
              swap(arr, i, largest);
              heapify(arr, n, largest);
          }
      }
  
      private static void swap(int[] arr, int i, int j) {
          int temp = arr[i];
          arr[i] = arr[j];
          arr[j] = temp;
      }
  }
  ```
