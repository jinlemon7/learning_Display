# 取模
  a % (2^n) = a & (2^n - 1)  
  - 这种方式除数只适用于2^n, 不适用于10，7等

  例：9 % 4 = 9 & 3 = (1001)<sub>2</sub> & (011)<sub>2</sub> = 1  

# 判断奇偶性
  奇数：a & 1 = 1  
  偶数：a & 1 = 0  

# 数字翻倍/减半
  有符号左移右移( <<, >> )    负数右移高位补1  
  `>>>` 为无符号右移，负数右移高位补0  

  翻倍：a << 1  
  减半：a >> 1  

# 交换两数
  a = a ^ b;
  b = a ^ b;    // a ^ b ^ b = a ^ 0 = a  
  a = a ^ b;    // a ^ b ^ a = b ^ 0 = b
